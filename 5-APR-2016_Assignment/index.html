Q2 > Create a function named beCheerful(). Within this function, console.log “good morning!” Call this function 84 times.

Soln: 

        function beCheerful(){
          for (var i = 1; i <= 84; i++) {
            console.log("good morning");
          }
        }

Q3 > Print all integers from -52 to 1066, with FOR loop.

Soln: 

        function printInt(){
          var numArr = [];
          
          for (var i = -52; i <= 1066; i++) {
            numArr.push(i);
          }  
        }

Q4 > For [1,3,5,7,9,13], print values that are greater than 2nd value.

Soln: 

        function printGreater(){
          var numArr = [1,3,5,7,9,13];
          var secEle = numArr[1];
          var result = [];

          for (var i = 0; i < numArr.length; i++){
            if (numArr[i] > secEle) {
              result.push(numArr[i]);
            }
          }
        }

Q6 > Print integers 2000-5280 inclusive, with WHILE.

Soln: 

        function whilePrint(){
          var i      = 2000;
          var numArr = [];

          while (i <= 5280) {
            numArr.push(i);
            i++;
          }
        }
