Q1 Construct a function to accept four parameters. Print all multiples of param1 from param2 to param3 inclusive unless multiple is equal to param4. Use a WHILE loop.

          function q1(num, ll, ul, ex){
            if ((ll < num) && (ul < num)){
              console.log("The range limits must be > Number");
            } else if (ll > ul){
              for (var i = ll; i >= ul; i--){
                if (((i % num) === 0) && (i !== ex)){
                  console.log(i);
                }
              }
            } else if (ul >= ll){
              for (var i = ll; i <= ul; i++){
                if (((i % num) === 0) && (i !== ex)){
                  console.log(i);
                }
              }
            }
          }

Q3 Create a function that accepts a number as an input. Return an array that counts down by one, from the number (as array’s ‘zero’th element) down to 0 (as the last element). How long is this array?

          function q3arr(num){
            var arr = [];
            for (var i = num; i >= 0; i--){
              arr.push(i);
            }
            return arr;
          }
